Person.java

package com.mycompany.framepersonalizado.model;

import java.io.Serializable;

/**
 *
 * @author a19luisdr
 */
public class Person implements Serializable
{

    private static final long serialVersionUID = 1234L;

    public static int count = 0;

    private int id;

    private String name;

    private String occupation;

    private AgeCategory ageCategory;

    private EmploymentCategory empCat;

    private String taxId;

    private boolean usCitizen;

    private Gender gender;

    public Person(String name, String occupation, AgeCategory ageCategory, EmploymentCategory empCat, String taxId, boolean usCitizen, Gender gender)
    {
        id = Person.count++;
        this.name = name;
        this.occupation = occupation;
        this.ageCategory = ageCategory;
        this.empCat = empCat;
        this.taxId = taxId;
        this.usCitizen = usCitizen;
        this.gender = gender;
    }

    public void setName(String name)
    {
        this.name = name;
    }

    public void setOccupation(String occupation)
    {
        this.occupation = occupation;
    }

    public void setAgeCategory(AgeCategory ageCategory)
    {
        this.ageCategory = ageCategory;
    }

    public void setEmpCat(EmploymentCategory empCat)
    {
        this.empCat = empCat;
    }

    public void setTaxId(String taxId)
    {
        this.taxId = taxId;
    }

    public void setUsCitizen(boolean usCitizen)
    {
        this.usCitizen = usCitizen;
    }

    public void setGender(Gender gender)
    {
        this.gender = gender;
    }

    public static int getCount()
    {
        return count;
    }

    public int getId()
    {
        return id;
    }

    public String getName()
    {
        return name;
    }

    public String getOccupation()
    {
        return occupation;
    }

    public AgeCategory getAgeCategory()
    {
        return ageCategory;
    }

    public EmploymentCategory getEmpCat()
    {
        return empCat;
    }

    public String getTaxId()
    {
        return taxId;
    }

    public boolean isUsCitizen()
    {
        return usCitizen;
    }

    public Gender getGender()
    {
        return gender;
    }
}

Database.java

package com.mycompany.framepersonalizado.model;

import java.io.File;
import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.IOException;
import java.io.ObjectInputStream;
import java.io.ObjectOutputStream;
import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.sql.Statement;
import java.util.Collections;
import java.util.LinkedList;
import java.util.List;

/**
 *
 * @author a19luisdr
 */
public class Database
{

    private int port;
    private String user;
    private String password;

    private Connection con;

    public void configure(int port, String user, String password)
    {
        this.port = port;
        this.user = user;
        this.password = password;
    }

    private void connect() throws Exception
    {
        con = DriverManager.getConnection("jdbc:mysql://10.21.0.142:" + port + "/people_di_db?useSSL=false", user, password);
    }

    private void disconnect() throws Exception
    {
        con.close();
    }

    private List<Person> people;

    public Database()
    {
        people = new LinkedList<Person>();
    }

    public void addPerson(Person person)
    {
        people.add(person);
    }

    public List<Person> getPeople()
    {
        return Collections.unmodifiableList(people);
    }

    public void removePerson(int index)
    {
        people.remove(index);
    }

    public void saveToFile(File file) throws IOException
    {
        FileOutputStream fos = new FileOutputStream(file);

        ObjectOutputStream oos = new ObjectOutputStream(fos);

        oos.writeObject(people);

        oos.close();
        fos.close();
    }

    public void loadFromFile(File file) throws IOException
    {
        FileInputStream fis = new FileInputStream(file);

        ObjectInputStream ois = new ObjectInputStream(fis);

        try
        {
            people.clear();
            people.addAll((LinkedList<Person>) ois.readObject());

            if (!people.isEmpty())
            {
                Person.count = people.get(people.size() - 1).getId() + 1;
            } else
            {
                Person.count = 0;
            }
        } catch (ClassNotFoundException e)
        {
            e.printStackTrace();
        }

        ois.close();
        fis.close();
    }

    public void saveToDatabase() throws Exception
    {
        connect();

        con.setAutoCommit(false);

        Statement deleteStatement = con.createStatement();

        deleteStatement.executeUpdate("DELETE FROM people");

        for (Person person : people)
        {
            PreparedStatement insertStatement = con.prepareStatement("INSERT INTO people VALUES(?, ?, ?, ?, ?, ?, ?, ?)");

            insertStatement.setInt(1, person.getId());
            insertStatement.setString(2, person.getName());
            insertStatement.setString(3, person.getAgeCategory().name());
            insertStatement.setString(4, person.getEmpCat().name());
            insertStatement.setString(5, person.getTaxId());
            insertStatement.setBoolean(6, person.isUsCitizen());
            insertStatement.setString(7, person.getGender().name());
            insertStatement.setString(8, person.getOccupation());

            insertStatement.executeUpdate();
        }

        con.commit();

        disconnect();
    }

    public void loadFromDatabase() throws Exception
    {
        connect();

        people.clear();

        Statement selectStatement = con.createStatement();

        ResultSet resultado = selectStatement.executeQuery("SELECT * FROM people ORDER BY id");

        while (resultado.next())
        {
            people.add(new Person(resultado.getString("name"), resultado.getString("occupation"), AgeCategory.valueOf(resultado.getString("age")), EmploymentCategory.valueOf(resultado.getString("employment_status")), resultado.getString("tax_id"), resultado.getBoolean("us_citizen"), Gender.valueOf(resultado.getString("gender"))));
        }

        disconnect();
    }
}
