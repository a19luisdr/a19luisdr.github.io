package com.mycompany.framepersonalizado.gui;

import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.awt.event.ItemEvent;
import java.awt.event.ItemListener;
import java.awt.event.KeyEvent;
import javax.swing.JCheckBoxMenuItem;
import javax.swing.JFileChooser;
import javax.swing.JMenu;
import javax.swing.JMenuBar;
import javax.swing.JMenuItem;
import javax.swing.JOptionPane;
import javax.swing.KeyStroke;

/**
 *
 * @author a19luisdr
 */
public class BarraMenu extends JMenuBar
{

    private JMenu fileMenu;
    private JMenuItem exportDataItem;
    private JMenuItem importDataItem;
    private JMenuItem exportDataToDatabaseItem;
    private JMenuItem exitItem;

    private JMenu windowMenu;
    private JMenu showMenu;
    private JCheckBoxMenuItem showPersonFormItem;
    private JMenuItem preferencesItem;

    private ShowWindowListener showWindowListener;

    private ShowPreferencesDialogListener showPreferencesDialogListener;

    private int dividerLocation;

    private JFileChooser fileChooser;

    public BarraMenu()
    {
        fileMenu = new JMenu("File");
        exportDataItem = new JMenuItem("Export data...");
        importDataItem = new JMenuItem("Import data...");
        exportDataToDatabaseItem = new JMenuItem("Export data to database...");
        exitItem = new JMenuItem("Exit");

        fileChooser = new JFileChooser();

        fileChooser.setFileFilter(new PersonFileFilter());

        exportDataItem.setAccelerator(KeyStroke.getKeyStroke(KeyEvent.VK_E, ActionEvent.CTRL_MASK));
        importDataItem.setAccelerator(KeyStroke.getKeyStroke(KeyEvent.VK_I, ActionEvent.CTRL_MASK));
        exportDataToDatabaseItem.setAccelerator(KeyStroke.getKeyStroke(KeyEvent.VK_D, ActionEvent.CTRL_MASK));

        exitItem.setAccelerator(KeyStroke.getKeyStroke(KeyEvent.VK_X, ActionEvent.CTRL_MASK));

        exitItem.addActionListener(new ActionListener()
        {
            @Override
            public void actionPerformed(ActionEvent ae)
            {
                int resultado = JOptionPane.showConfirmDialog(BarraMenu.this, "Do you want to close the application?", "Confirmation", JOptionPane.OK_CANCEL_OPTION);

                if (resultado == 0)
                {
                    System.exit(0);
                }
            }
        });

        fileMenu.add(exportDataItem);
        fileMenu.add(importDataItem);
        fileMenu.addSeparator();
        fileMenu.add(exportDataToDatabaseItem);
        fileMenu.addSeparator();
        fileMenu.add(exitItem);

        add(fileMenu);

        windowMenu = new JMenu("Window");
        showMenu = new JMenu("Show");
        showPersonFormItem = new JCheckBoxMenuItem("Person Form");

        showPersonFormItem.setSelected(true);

        showPersonFormItem.addItemListener(new ItemListener()
        {
            @Override
            public void itemStateChanged(ItemEvent ie)
            {
                if (ie.getStateChange() == ItemEvent.SELECTED)
                {
                    showWindowListener.setPersonFormVisible(dividerLocation);
                } else
                {
                    dividerLocation = showWindowListener.getDividerLocation();

                    showWindowListener.setPersonFormInvisible();
                }
            }
        });

        showMenu.add(showPersonFormItem);

        preferencesItem = new JMenuItem("Preferences...");

        preferencesItem.addActionListener(new ActionListener()
        {
            @Override
            public void actionPerformed(ActionEvent ae)
            {
                showPreferencesDialogListener.showPreferencesDialog();
            }
        });

        preferencesItem.setAccelerator(KeyStroke.getKeyStroke(KeyEvent.VK_P, ActionEvent.CTRL_MASK));

        windowMenu.add(showMenu);
        windowMenu.add(preferencesItem);

        add(windowMenu);
    }

    public void setShowWindowListener(ShowWindowListener showWindowListener)
    {
        this.showWindowListener = showWindowListener;
    }

    public void setOpenOrSaveListener(OpenOrSaveListener openOrSaveListener)
    {
        exportDataItem.addActionListener(openOrSaveListener.getSaveActionListener(fileChooser));

        importDataItem.addActionListener(openOrSaveListener.getOpenActionListener(fileChooser));

        exportDataToDatabaseItem.addActionListener(openOrSaveListener.getSaveToDatabaseActionListener());
    }

    public void setShowPreferencesDialogListener(ShowPreferencesDialogListener showPreferencesDialogListener)
    {
        this.showPreferencesDialogListener = showPreferencesDialogListener;
    }
}

        normasDelLayout.gridy = 1;

        add(etiquetaPassword, normasDelLayout);

        normasDelLayout.gridy = 2;

        add(etiquetaPort, normasDelLayout);

        normasDelLayout.anchor = GridBagConstraints.WEST;
        normasDelLayout.gridx = 1;
        normasDelLayout.gridy = 0;

        add(userTextField, normasDelLayout);

        normasDelLayout.gridy = 1;

        add(passwordTextField, normasDelLayout);

        normasDelLayout.gridy = 2;

        add(portSpinner, normasDelLayout);

    }

    {
        return userTextField.getText();
    }

    public String getPassword()
    {
        return passwordTextField.getText();
    }

    {
        combo = new JComboBox();

        DefaultComboBoxModel modeloEmployment = new DefaultComboBoxModel();
        modeloEmployment.addElement(EmploymentCategory.EMPLOYED);
        modeloEmployment.addElement(EmploymentCategory.SELFEMPLOYED);

    }

    {
        combo.setSelectedItem(value);

        combo.addActionListener(new ActionListener()
        {
            @Override
            public void actionPerformed(ActionEvent ae)
            {
                fireEditingStopped();
            }
        });

        return combo;
    }

package com.mycompany.framepersonalizado.gui;

import com.mycompany.framepersonalizado.controller.Controller;
import java.awt.BorderLayout;
import java.awt.Dimension;
import java.awt.HeadlessException;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.io.File;
import java.io.IOException;
import javax.swing.JFileChooser;
import javax.swing.JFrame;
import javax.swing.JOptionPane;
import javax.swing.JSplitPane;
import javax.swing.JTabbedPane;
import javax.swing.JTextArea;
import javax.swing.WindowConstants;

/**
 *
 * @author a19luisdr
 */
public class MainFrame extends JFrame
{

    private Controller controller;

    private BarraMenu menuBar;

    private Toolbar toolbar;
    private FormPanel formulario;
    private TablePanel tablePanel;
    private JTextArea textArea;
    private JSplitPane splitPane;

    private PreferencesDialog preferencesDialog;

    private Dimension frameSize;
    private Dimension formularioPreferredSize;

    private JTabbedPane tabbedPane;

    public MainFrame(String title) throws HeadlessException
    {
        super(title);

        frameSize = new Dimension(800, 600);

        setSize(frameSize);

        setLocationRelativeTo(null);

        setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);

        setLayout(new BorderLayout());

        formularioPreferredSize = new Dimension(250, getHeight());

        controller = new Controller();

        configureDatabase();

        loadInformationFromDatabase();

        formulario = new FormPanel();
        formulario.setFormListener(new FormListener()
        {
            @Override
            public void infoEmited(FormEvent e)
            {
                controller.addPerson(e);

                tablePanel.refresh();
            }
        });
        formulario.setPreferredSize(formularioPreferredSize);

        tablePanel = new TablePanel();
        tablePanel.setData(controller.getPersonas());
        tablePanel.setPersonTableListener(new PersonTableListener()
        {
            @Override
            public void rowDeleted(int row)
            {
                controller.removePerson(row);
            }
        });

        textArea = new JTextArea();

        tabbedPane = new JTabbedPane();

        tabbedPane.addTab("Table panel", tablePanel);

        tabbedPane.addTab("Text area", textArea);

        splitPane = new JSplitPane(JSplitPane.HORIZONTAL_SPLIT, formulario, tabbedPane);

        add(splitPane, BorderLayout.CENTER);

        OpenOrSaveListener openOrSaveListener = new OpenOrSaveListener()
        {
            @Override
            public ActionListener getOpenActionListener(JFileChooser fileChooser)
            {
                return new ActionListener()
                {
                    @Override
                    public void actionPerformed(ActionEvent ae)
                    {
                        int resultado = fileChooser.showOpenDialog(getParent());

                        if (resultado == JFileChooser.APPROVE_OPTION)
                        {
                            try
                            {
                                File file;

                                File selectedFile = fileChooser.getSelectedFile();

                                String extension = Utils.getExtension(selectedFile);

                                if (extension == null)
                                {
                                    file = new File(selectedFile.getAbsolutePath() + ".per");
                                } else
                                {
                                    file = selectedFile;
                                }

                                controller.loadFromFile(file);

                                tablePanel.refresh();
                            } catch (IOException e)
                            {
                                JOptionPane.showMessageDialog(getParent(), "An error occurred while importing the file.", "Import data failed", JOptionPane.ERROR_MESSAGE);
                            }
                        }
                    }
                };
            }

            @Override
            public ActionListener getSaveActionListener(JFileChooser fileChooser)
            {
                return new ActionListener()
                {
                    @Override
                    public void actionPerformed(ActionEvent ae)
                    {
                        int resultado = fileChooser.showSaveDialog(getParent());

                        if (resultado == JFileChooser.APPROVE_OPTION)
                        {
                            try
                            {
                                File file;

                                File selectedFile = fileChooser.getSelectedFile();

                                String extension = Utils.getExtension(selectedFile);

                                if (extension == null)
                                {
                                    file = new File(selectedFile.getAbsolutePath() + ".per");
                                } else
                                {
                                    file = selectedFile;
                                }

                                controller.saveToFile(file);
                            } catch (IOException e)
                            {
                                JOptionPane.showMessageDialog(getParent(), "An error occurred while exporting the file.", "Export data failed", JOptionPane.ERROR_MESSAGE);
                            }
                        }
                    }
                };
            }

            @Override
            public ActionListener getSaveToDatabaseActionListener()
            {
                return new ActionListener()
                {
                    @Override
                    public void actionPerformed(ActionEvent ae)
                    {
                        saveInformationToDatabase();
                    }
                };
            }
        };

        toolbar = new Toolbar();

        toolbar.setOpenOrSaveListener(openOrSaveListener);

        add(toolbar, BorderLayout.NORTH);

        menuBar = new BarraMenu();

        menuBar.setShowWindowListener(new ShowWindowListener()
        {
            @Override
            public void setPersonFormVisible(int dividerLocation)
            {
                formulario.setVisible(true);

                splitPane.setDividerLocation(dividerLocation);
            }

            @Override
            public void setPersonFormInvisible()
            {
                formulario.setVisible(false);
            }

package com.mycompany.framepersonalizado.gui;

import com.mycompany.framepersonalizado.model.EmploymentCategory;
import com.mycompany.framepersonalizado.model.Person;
import java.util.List;
import javax.swing.table.AbstractTableModel;

/**
 *
 * @author a19luisdr
 */
public class PersonTableModel extends AbstractTableModel
{

    private List<Person> personList;

    private String[] columnNames =
    {
        "ID", "Name", "Occupation", "Age Category",
        "Employment Category", "US Citizen", "Tax ID", "Gender"
    };

    public void setData(List<Person> personList)
    {
        this.personList = personList;
    }

    @Override
    public int getRowCount()
    {
        return personList.size();
    }

    @Override
    public int getColumnCount()
    {
        return columnNames.length;
    }

    @Override
    public Object getValueAt(int i, int i1)
    {
        switch (i1)
        {
            case 0:
                return personList.get(i).getId();
            case 1:
                return personList.get(i).getName();
            case 2:
                return personList.get(i).getOccupation();
            case 3:
                return personList.get(i).getAgeCategory();
            case 4:
                return personList.get(i).getEmpCat();
            case 5:
                return personList.get(i).isUsCitizen();
            case 6:
                return personList.get(i).getTaxId();
            case 7:
                return personList.get(i).getGender();
            default:
                return null;
        }
    }

    @Override
    public String getColumnName(int column)
    {
        if (column < 0 || column >= getColumnCount())
        {
            throw new ArrayIndexOutOfBoundsException(column);
        } else
        {
            return columnNames[column];
        }
    }

    @Override
    public boolean isCellEditable(int row, int column)
    {
        switch (column)
        {
            case 1:
                return true;
            case 4:
                return true;
            case 5:
                return true;
            case 6:
                return personList.get(row).isUsCitizen();
            default:
                return false;
        }
    }

    @Override
    public void setValueAt(Object aValue, int row, int column)
    {
        switch (column)
        {
            case 1:
                personList.get(row).setName((String) aValue);
                break;
            case 4:
                personList.get(row).setEmpCat((EmploymentCategory) aValue);
                break;
            case 5:
                personList.get(row).setUsCitizen((Boolean) aValue);
                break;
            case 6:
                personList.get(row).setTaxId((String) aValue);
            default:
                break;
        }
    }

    @Override
    public Class<?> getColumnClass(int columnIndex)
    {
        switch (columnIndex)
        {
            case 0:
                return Integer.class;
            case 1:
                return String.class;
            case 2:
                return String.class;
            case 3:
                return String.class;
            case 4:
                return EmploymentCategory.class;
            case 5:
                return Boolean.class;
            case 6:
                return String.class;
            case 7:
                return String.class;
            default:
                return null;
        }
    }
}

package com.mycompany.framepersonalizado.gui;

import com.mycompany.framepersonalizado.model.EmploymentCategory;
import com.mycompany.framepersonalizado.model.Person;
import java.awt.BorderLayout;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.awt.event.MouseAdapter;
import java.awt.event.MouseEvent;
import java.util.List;
import javax.swing.JMenuItem;
import javax.swing.JPanel;
import javax.swing.JPopupMenu;
import javax.swing.JScrollPane;
import javax.swing.JTable;

/**
 *
 * @author a19luisdr
 */
public class TablePanel extends JPanel
{

    private PersonTableListener personTableListener;

    private JTable table;
    private PersonTableModel tableModel;

    private JPopupMenu popup;
    private JMenuItem itemRemovePerson;

    public TablePanel()
    {
        tableModel = new PersonTableModel();
        table = new JTable(tableModel);
        setLayout(new BorderLayout());

        popup = new JPopupMenu();

        itemRemovePerson = new JMenuItem("Remove person");

        itemRemovePerson.addActionListener(new ActionListener()
        {
            @Override
            public void actionPerformed(ActionEvent ae)
            {
                if (personTableListener != null)
                {
                    int row = table.getSelectedRow();

                    personTableListener.rowDeleted(row);

                    tableModel.fireTableRowsDeleted(row, row);
                }
            }
        });

        popup.add(itemRemovePerson);

        table.setRowHeight(25);

        table.setDefaultRenderer(EmploymentCategory.class, new EmploymentCategoryRenderer());

        table.setDefaultEditor(EmploymentCategory.class, new EmploymentCategoryEditor());

        table.addMouseListener(new MouseAdapter()
        {
            public void mousePressed(MouseEvent e)
            {
                maybeShowPopup(e);
            }

            public void mouseReleased(MouseEvent e)
            {
                maybeShowPopup(e);
            }

            private void maybeShowPopup(MouseEvent e)
            {
                int row = table.rowAtPoint(e.getPoint());

                if (row >= 0 && row < table.getRowCount())
                {
                    table.setRowSelectionInterval(row, row);
                } else
                {
                    table.clearSelection();
                }

                if (e.isPopupTrigger() && table.getSelectedRow() != -1)
                {
                    popup.show(e.getComponent(), e.getX(), e.getY());
                }
            }
        });

        table.getSelectedRow();

        add(new JScrollPane(table), BorderLayout.CENTER);
    }

    public void setData(List<Person> personList)
    {
        tableModel.setData(personList);
    }

    public void refresh()
    {
        tableModel.fireTableDataChanged();
    }

    public void setPersonTableListener(PersonTableListener personTableListener)
    {
        this.personTableListener = personTableListener;
    }
}

Utils.java

package com.mycompany.framepersonalizado.gui;

import java.io.File;
import javax.swing.ImageIcon;

/**
 *
 * @author a19luisdr
 */
public class Utils
{

    public final static String jpeg = "jpeg";
    public final static String jpg = "jpg";
    public final static String gif = "gif";
    public final static String tiff = "tiff";
    public final static String tif = "tif";
    public final static String png = "png";

    /*
     * Get the extension of a file.
     */
    public static String getExtension(File f)
    {
        String ext = null;
        String s = f.getName();
        int i = s.lastIndexOf('.');

        if (i > 0 && i < s.length() - 1)
        {
            ext = s.substring(i + 1).toLowerCase();
        }
        return ext;
    }

    /**
     * Returns an ImageIcon, or null if the path was invalid.
     */
    protected static ImageIcon createImageIcon(String path)
    {
        java.net.URL imgURL = Utils.class.getResource(path);
        if (imgURL != null)
        {
            return new ImageIcon(imgURL);
        } else
        {
            System.err.println("Couldn't find file: " + path);
            return null;
        }
    }
}
