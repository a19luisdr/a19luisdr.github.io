# Ejemplo controlador

Controller.java

package com.mycompany.framepersonalizado.controller;

import com.mycompany.framepersonalizado.gui.FormEvent;
import com.mycompany.framepersonalizado.gui.PreferencesEvent;
import com.mycompany.framepersonalizado.model.AgeCategory;
import com.mycompany.framepersonalizado.model.Database;
import com.mycompany.framepersonalizado.model.EmploymentCategory;
import com.mycompany.framepersonalizado.model.Gender;
import com.mycompany.framepersonalizado.model.Person;
import java.io.File;
import java.io.IOException;
import java.util.List;
import java.util.prefs.Preferences;

/**
 *
 * @author a19luisdr
 */
public class Controller
{

    private Database db = new Database();

    private Preferences prefs = Preferences.userRoot().node(this.getClass().getName());

    private final String ID0 = "PORT";
    private final String ID1 = "USER";
    private final String ID2 = "PASSWORD";

    public void addPerson(FormEvent ev)
    {
        String name = ev.getName();

        String occupation = ev.getOccupation();

        int ageCatId = ev.getAgeCategory();

        String empCat = ev.getEmploymentCategory();

        String taxId = ev.getTaxID();

        boolean isUs = ev.getUsCitizen();

        String gender = ev.getGender();

        AgeCategory ageCategory = null;

        switch (ageCatId)
        {
            case 0:
                ageCategory = AgeCategory.CHILD;
                break;
            case 1:
                ageCategory = AgeCategory.ADULT;
                break;
            case 2:
                ageCategory = AgeCategory.SENIOR;
                break;
        }

        EmploymentCategory empCategory;

        switch (empCat)
        {
            case "employed":
                empCategory = EmploymentCategory.EMPLOYED;
                break;
            case "self-employed":
                empCategory = EmploymentCategory.SELFEMPLOYED;
                break;
            case "unemployed":
                empCategory = EmploymentCategory.UNEMPLOYED;
                break;
            default:
                empCategory = EmploymentCategory.OTHER;
                break;
        }

        Gender genderCat;

        if (gender.equals("male"))
        {
            genderCat = Gender.MALE;
        } else
        {
            genderCat = Gender.FEMALE;
        }

        Person person = new Person(name, occupation, ageCategory, empCategory, taxId, isUs, genderCat);

        db.addPerson(person);
    }

    public List<Person> getPersonas()
    {
        return db.getPeople();
    }

    public void removePerson(int index)
    {
        db.removePerson(index);
    }

    public void saveToFile(File file) throws IOException
    {
        db.saveToFile(file);
    }

    public void loadFromFile(File file) throws IOException
    {
        db.loadFromFile(file);
    }

    public void saveToDatabase() throws Exception
    {
        db.saveToDatabase();
    }

    public void loadFromDatabase() throws Exception
    {
        db.loadFromDatabase();
    }

    public void setPreferences(PreferencesEvent preferencesEvent)
    {
        prefs.putInt(ID0, preferencesEvent.getPort());
        prefs.put(ID1, preferencesEvent.getUser());
        prefs.put(ID2, preferencesEvent.getPassword());
    }

    public PreferencesEvent getPreferences()
    {
        return new PreferencesEvent(prefs.getInt(ID0, 3306), prefs.get(ID1, ""), prefs.get(ID2, ""));
    }

    public void configure(int port, String user, String password)
    {
        db.configure(port, user, password);
    }
}
